//Realizar un programa que muestre la cantidad de números que son múltiplos de 2 o de 3
//comprendidos entre 1 y 100.

Algoritmo cuadroNumero
	
	Definir i Como Entero
	Definir multiplosDeDosOTres Como Entero
	
	multiplosDeDosOTres <- 0
	
	Para i <- 1 Hasta 100 Con Paso 1 Hacer
		Escribir i , " es múltiplo de 2 o 3: " , i MOD 2 = 0 O i MOD 3 = 0
		
		Si i MOD 2 = 0 O i MOD 3 = 0
			multiplosDeDosOTres <- multiplosDeDosOTres + 1
		FinSi
	Fin Para
	
	Escribir "La cantidad de múltiplos de 2 o 3 comprendidos de 1 a 100 es: " , multiplosDeDosOTres
	
FinAlgoritmo
