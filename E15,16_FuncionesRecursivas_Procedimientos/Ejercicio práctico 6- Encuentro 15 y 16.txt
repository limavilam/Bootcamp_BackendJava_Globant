SubProceso letters(letra Por Valor)
	
	//Uso de operadores relacionales con letras y cadenas.
	Si letra >= "M" Y letra <= "T" Entonces
        Escribir "La letra está entre M y T"
    Sino

    FinSi	
	
FinSubProceso
